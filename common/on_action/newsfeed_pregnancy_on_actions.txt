on_pregnancy_mother = {
    on_actions = {
        newsfeed_on_pregnancy_mother
    }
}

newsfeed_on_pregnancy_mother = {
    trigger = {
        exists = scope:mother
    }

    effect = {
        every_player = {
            limit = {
                OR = {
                    newsfeed_is_interested_in_pregnancy = { PERSON = scope:mother }
                    AND = {
                        exists = scope:father
                        newsfeed_is_interested_in_pregnancy = { PERSON = scope:father }
                    }
                }
            }

            newsfeed_notify_pregnancy_effect = yes
        }
    }
}
